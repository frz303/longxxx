from flask import Flask, render_template, redirect, url_for, request, flash from flask_sqlalchemy import SQLAlchemy from datetime import datetime, date, timedelta from sqlalchemy import func import os app = Flask(__name__) # 为简单起见，线上直接用固定密钥（你可以改成别的字符串） app.config['SECRET_KEY'] = 'change-this-key' # 数据库使用项目目录下的 farm.db 文件 BASE_DIR = os.path.dirname(os.path.abspath(__file__)) db_path = os.path.join(BASE_DIR, 'farm.db') app.config['SQLALCHEMY_DATABASE_URI'] = f'sqlite:///{db_path}' app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False db = SQLAlchemy(app) # ========== 模型 ========== class Plot(db.Model): id = db.Column(db.Integer, primary_key=True) name = db.Column(db.String(100), nullable=False) area = db.Column(db.Float, nullable=True) description = db.Column(db.String(255), nullable=True) feedings = db.relationship('Feeding', backref='plot', lazy=True) incomes = db.relationship('Income', backref='plot', lazy=True) class Feeding(db.Model): id = db.Column(db.Integer, primary_key=True) plot_id = db.Column(db.Integer, db.ForeignKey('plot.id'), nullable=False) date = db.Column(db.Date, nullable=False, default=date.today) feed_type = db.Column(db.String(100), nullable=False) amount = db.Column(db.Float, nullable=True) note = db.Column(db.String(255), nullable=True) class Income(db.Model): id = db.Column(db.Integer, primary_key=True) plot_id = db.Column(db.Integer, db.ForeignKey('plot.id'), nullable=False) date = db.Column(db.Date, nullable=False, default=date.today) yield_weight = db.Column(db.Float, nullable=True) # kg income = db.Column(db.Float, nullable=True) # 元 size_grade = db.Column(db.String(50), nullable=True) note = db.Column(db.String(255), nullable=True) class CustomMetric(db.Model): id = db.Column(db.Integer, primary_key=True) name = db.Column(db.String(100), nullable=False) value = db.Column(db.Float, nullable=True) unit = db.Column(db.String(20), nullable=True) note = db.Column(db.String(255), nullable=True) # ========== 首页总看板 ========== @app.route("/") def index(): # 按地块统计总产量和总收入 plot_stats = ( db.session.query( Plot.id, Plot.name, Plot.area, func.sum(Income.yield_weight).label("total_yield"), func.sum(Income.income).label("total_income"), ) .outerjoin(Income, Income.plot_id == Plot.id) .group_by(Plot.id, Plot.name, Plot.area) .all() ) # 最近 30 天投喂总量（用于图表） today = date.today() start_date = today - timedelta(days=30) feeding_data = ( db.session.query( Feeding.date, func.sum(Feeding.amount).label("total_amount") ) .filter(Feeding.date >= start_date) .group_by(Feeding.date) .order_by(Feeding.date) .all() ) metrics = CustomMetric.query.all() feeding_labels = [f.date.strftime("%m-%d") for f in feeding_data] feeding_values = [float(f.total_amount or 0) for f in feeding_data] return render_template( "index.html", plot_stats=plot_stats, metrics=metrics, feeding_labels=feeding_labels, feeding_values=feeding_values, ) # ========== 地块管理 ========== @app.route("/plots") def plots_list(): plots = Plot.query.all() return render_template("plots.html", plots=plots) @app.route("/plots/create", methods=["GET", "POST"]) def plots_create(): if request.method == "POST": name = request.form.get("name") area = request.form.get("area") or None description = request.form.get("description") if not name: flash("地块名称不能为空") return redirect(url_for("plots_create")) plot = Plot(name=name, area=area, description=description) db.session.add(plot) db.session.commit() return redirect(url_for("plots_list")) return render_template("plots_form.html", plot=None) @app.route("/plots//edit", methods=["GET", "POST"]) def plots_edit(plot_id): plot = Plot.query.get_or_404(plot_id) if request.method == "POST": plot.name = request.form.get("name") plot.area = request.form.get("area") or None plot.description = request.form.get("description") db.session.commit() return redirect(url_for("plots_list")) return render_template("plots_form.html", plot=plot) @app.route("/plots//delete", methods=["POST"]) def plots_delete(plot_id): plot = Plot.query.get_or_404(plot_id) db.session.delete(plot) db.session.commit() return redirect(url_for("plots_list")) #
